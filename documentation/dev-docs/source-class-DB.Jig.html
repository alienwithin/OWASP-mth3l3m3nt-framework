 25 //! In-memory/flat-file DB wrapper
 26 class Jig {
 27 
 28     //@{ Storage formats
 29     const
 30         FORMAT_JSON=0,
 31         FORMAT_Serialized=1;
 32     //@}
 33 
 34     protected
 35         //! UUID
 36         $uuid,
 37         //! Storage location
 38         $dir,
 39         //! Current storage format
 40         $format,
 41         //! Jig log
 42         $log,
 43         //! Memory-held data
 44         $data;
 45 
 46     /**
 47     *   Read data from memory/file
 48     *   @return array
 49     *   @param $file string
 50     **/
 51     function read($file) {
 52         if (!$this->dir)
 53             return isset($this->data[$file])?$this->data[$file]:array();
 54         $fw=\Base::instance();
 55         if (!is_file($dst=$this->dir.$file))
 56             return array();
 57         $raw=$fw->read($dst);
 58         switch ($this->format) {
 59             case self::FORMAT_JSON:
 60                 $data=json_decode($raw,TRUE);
 61                 break;
 62             case self::FORMAT_Serialized:
 63                 $data=$fw->unserialize($raw);
 64                 break;
 65         }
 66         return $data;
 67     }
 68 
 69     /**
 70     *   Write data to memory/file
 71     *   @return int
 72     *   @param $file string
 73     *   @param $data array
 74     **/
 75     function write($file,array $data=NULL) {
 76         if (!$this->dir)
 77             return count($this->data[$file]=$data);
 78         $fw=\Base::instance();
 79         switch ($this->format) {
 80             case self::FORMAT_JSON:
 81                 $out=json_encode($data,@constant('JSON_PRETTY_PRINT'));
 82                 break;
 83             case self::FORMAT_Serialized:
 84                 $out=$fw->serialize($data);
 85                 break;
 86         }
 87         return $fw->write($this->dir.'/'.$file,$out);
 88     }
 89

 91     *   Return directory
 92     *   @return string
 93     **/

 96     }

124     /**
125     *   Clean storage
126     *   @return NULL
127     **/
128     function drop() {
129         if (!$this->dir)
130             $this->data=array();
131         elseif ($glob=@glob($this->dir.'/*',GLOB_NOSORT))
132             foreach ($glob as $file)
133                 @unlink($file);
134     }

141     function __construct($dir=NULL,$format=self::FORMAT_JSON) {
142         if ($dir && !is_dir($dir))
143             mkdir($dir,\Base::MODE,TRUE);
144         $this->uuid=\Base::instance()->hash($this->dir=$dir);
145         $this->format=$format;
