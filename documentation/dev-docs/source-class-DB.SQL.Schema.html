 293:     /**
 294:      * add a new column to this table
 295:      * @param string|Column $key column name or object
 296:      * @param null|array $args optional config array
 297:      * @return \DB\SQL\Column
 298:      */
 299:     public function addColumn($key,$args = null)
 300:     {
 301:         if ($key instanceof Column) {
 302:             $args = $key->getColumnArray();
 303:             $key = $key->name;
 304:         }
 305:         if (array_key_exists($key,$this->columns))
 306:             trigger_error(sprintf(self::TEXT_ColumnExists,$key));
 307:         $column = new Column($key, $this);
 308:         if ($args)
 309:             foreach ($args as $arg => $val)
 310:                 $column->{$arg} = $val;
 311:         // skip default pkey field
 312:         if (count($this->pkeys) == 1 && in_array($key,$this->pkeys))
 313:             return $column;
 314:         return $this->columns[$key] =& $column;

 317:     /**
 318:      * create index on one or more columns
 319:      * @param string|array $index_cols Column(s) to be indexed
 320:      * @param              $search_cols
 321:      * @param bool         $unique     Unique index
 322:      * @param int          $length     index length for text fields in mysql
 323:      */
 324:     protected function _addIndex($index_cols, $search_cols, $unique, $length)
 325:     {
 326:         if (!is_array($index_cols))
 327:             $index_cols = array($index_cols);
 328:         $quotedCols = array_map(array($this->db, 'quotekey'), $index_cols);
 329:         if (preg_match('/mysql/', $this->db->driver()))
 330:             foreach($quotedCols as $i=>&$col)
 331:                 if(strtoupper($search_cols[$index_cols[$i]]['type']) == 'TEXT')
 332:                     $col.='('.$length.')';
 333:         $cols = implode(',', $quotedCols);
 334:         $name = $this->db->quotekey($this->name.'___'.implode('__', $index_cols));
 335:         $table = $this->db->quotekey($this->name);
 336:         $index = $unique ? 'UNIQUE INDEX' : 'INDEX';
 337:         $cmd = array(
 338:             'pgsql|sqlite2?|ibm|mssql|sybase|dblib|odbc|sqlsrv' =>
 339:                 "CREATE $index $name ON $table ($cols);",
 340:             'mysql' => //ALTER TABLE is used because of MySQL bug #48875
 341:                 "ALTER TABLE $table ADD $index $name ($cols);",
 342:         );
 343:         $query = $this->findQuery($cmd);
 344:         $this->queries[] = $query;
 345:     }

 347:     /**
 348:      * set primary / composite key to table
 349:      * @param string|array $pkeys
 350:      * @return bool
 351:      */
 352:     public function primary($pkeys) {
 353:         if (empty($pkeys))
 354:             return false;
 355:         if (!is_array($pkeys))
 356:             $pkeys = array($pkeys);
 357:         // single pkey
 358:         $this->increments = $pkeys[0];
 359:         $this->pkeys = $pkeys;
 360:         // drop duplicate pkey definition
 361:         if (array_key_exists($this->increments,$this->columns))
 362:             unset($this->columns[$this->increments]);
 363:         // set flag on new fields
 364:         foreach ($pkeys as $name)
 365:             if(array_key_exists($name,$this->columns))
 366:                 $this->columns[$name]->pkey = true;
 367:         // composite key
 368:         if (count($pkeys) > 1) {
 369:             $pkeys_quoted = array_map(array($this->db,'quotekey'), $pkeys);
 370:             $pk_string = implode(', ', $pkeys_quoted);
 371:             if (preg_match('/sqlite2?/', $this->db->driver())) {
 372:                 // rebuild table with new primary keys
 373:                 $this->rebuild_cmd['pkeys'] = $pkeys;
 374:                 return;
 375:             } else {
 376:                 $table = $this->db->quotekey($this->name);
 377:                 $table_key = $this->db->quotekey($this->name.'_pkey');
 378:                 $cmd = array(
 379:                     'odbc' =>
 380:                         "CREATE INDEX $table_key ON $table ( $pk_string );",
 381:                     'mysql' =>
 382:                         "ALTER TABLE $table DROP PRIMARY KEY, ADD PRIMARY KEY ( $pk_string );",
 383:                     'mssql|sybase|dblib|sqlsrv' => array(
 384:                         "ALTER TABLE $table DROP CONSTRAINT PK_".$this->name."_ID;",
 385:                         "ALTER TABLE $table ADD CONSTRAINT $table_key PRIMARY KEY ( $pk_string );",
 386:                     ),
 387:                     'pgsql' => array(
 388:                         "ALTER TABLE $table DROP CONSTRAINT $table_key;",
 389:                         "ALTER TABLE $table ADD CONSTRAINT $table_key PRIMARY KEY ( $pk_string );",
 390:                     ),
 391:                 );
 392:                 $query = $this->findQuery($cmd);
 393:                 if (!is_array($query))
 394:                     $query = array($query);
 395:                 foreach ($query as $q)
 396:                     $this->queries[] = $q;
 397:             }
 398:         }
 399:     }

 403: class TableCreator extends TableBuilder {
 404: 
 405:     const
 406:         TEXT_TableAlreadyExists = "Table `%s` already exists. Cannot create it.";
 407: 
 408:     /**
 409:      * generate SQL query for creating a basic table, containing an ID serial field
 410:      * and execute it if $exec is true, otherwise just return the generated query string
 411:      * @param bool $exec
 412:      * @return bool|TableModifier|string
 413:      */
 414:     public function build($exec = TRUE)
 415:     {
 416:         // check if already existing
 417:         if ($exec && in_array($this->name, $this->schema->getTables())) {
 418:             trigger_error(sprintf(self::TEXT_TableAlreadyExists,$this->name));
 419:             return false;
 420:         }
 421:         $cols = '';
 422:         if (!empty($this->columns))
 423:             foreach ($this->columns as $cname => $column) {
 424:                 // no defaults for TEXT type
 425:                 if ($column->default !== false && is_int(strpos(strtoupper($column->type),'TEXT'))) {
 426:                     trigger_error(sprintf(self::TEXT_NoDefaultForTEXT, $column->name));
 427:                     return false;
 428:                 }
 429:                 $cols .= ', '.$column->getColumnQuery();
 430:             }
 431:         $table = $this->db->quotekey($this->name);
 432:         $id = $this->db->quotekey($this->increments);
 433:         $cmd = array(
 434:             'sqlite2?|sybase|dblib' =>
 435:                 "CREATE TABLE $table ($id INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT".$cols.");",
 436:             'mysql' =>
 437:                 "CREATE TABLE $table ($id INTEGER NOT NULL PRIMARY KEY AUTO_INCREMENT".$cols.") DEFAULT CHARSET=utf8 COLLATE utf8_unicode_ci;",
 438:             'pgsql' =>
 439:                 "CREATE TABLE $table ($id SERIAL PRIMARY KEY".$cols.");",
 440:             'mssql|odbc|sqlsrv' =>
 441:                 "CREATE TABLE $table ($id INT IDENTITY CONSTRAINT PK_".$this->name."_ID PRIMARY KEY".$cols.");",
 442:             'ibm' =>
 443:                 "CREATE TABLE $table ($id INTEGER AS IDENTITY NOT NULL $cols, PRIMARY KEY($id));",
 444:         );
 445:         $query = $this->findQuery($cmd);
 446:         // composite key for sqlite
 447:         if (count($this->pkeys) > 1 && preg_match('/sqlite2?/', $this->db->driver())) {
 448:             $pk_string = implode(', ', $this->pkeys);
 449:             $query = "CREATE TABLE $table ($id INTEGER NULL".$cols.", PRIMARY KEY ($pk_string) );";
 450:             $newTable = new TableModifier($this->name, $this->schema);
 451:             // auto-incrementation in composite primary keys
 452:             $pk_queries = $newTable->_sqlite_increment_trigger($this->increments);
 453:             $this->queries = array_merge($this->queries, $pk_queries);
 454:         }
 455:         array_unshift($this->queries, $query);
 456:         // indexes
 457:         foreach ($this->columns as $cname => $column)
 458:             if ($column->index)
 459:                 $this->addIndex($cname, $column->unique);
 460:         if (!$exec)
 461:             return $this->queries;
 462:         $this->db->exec($this->queries);
 463:         return isset($newTable) ? $newTable : new TableModifier($this->name,$this->schema);
 464:     }
 465: 
 466:     /**
 467:      * create index on one or more columns
 468:      * @param string|array $columns Column(s) to be indexed
 469:      * @param bool         $unique  Unique index
 470:      * @param int          $length  index length for text fields in mysql
 471:      */
 472:     public function addIndex($columns, $unique = FALSE, $length = 20)
 473:     {
 474:         if (!is_array($columns))
 475:             $columns = array($columns);
 476:         $cols = $this->columns;
 477:         foreach ($cols as &$col)
 478:             $col = $col->getColumnArray();
 479:         parent::_addIndex($columns,$cols,$unique,$length);
 480:     }

 501:     {
 502:         // check if table exists
 503:         if (!in_array($this->name, $this->schema->getTables()))

 505: 
 506:         if ($sqlite = preg_match('/sqlite2?/', $this->db->driver())) {
 507:             $sqlite_queries = array();
 508:         }
 509:         $rebuild = false;
 510:         $additional_queries = $this->queries;
 511:         $this->queries = array();
 512:         // add new columns
 513:         foreach ($this->columns as $cname => $column) {
 514:             // not nullable fields should have a default value, when altering a table
 515:             if ($column->default === false && $column->nullable === false) {
 516:                 trigger_error(sprintf(self::TEXT_NotNullFieldNeedsDefault, $column->name));
 517:                 return false;
 518:             }
 519:             // no defaults for TEXT type
 520:             if($column->default !== false && is_int(strpos(strtoupper($column->type),'TEXT'))) {
 521:                 trigger_error(sprintf(self::TEXT_NoDefaultForTEXT, $column->name));
 522:                 return false;
 523:             }
 524:             $table = $this->db->quotekey($this->name);
 525:             $col_query = $column->getColumnQuery();
 526:             if ($sqlite) {
 527:                 // sqlite: dynamic column default only works when rebuilding the table
 528:                 if($column->default === Schema::DF_CURRENT_TIMESTAMP) {
 529:                     $rebuild = true;
 530:                     break;
 531:                 } else
 532:                     $sqlite_queries[] = "ALTER TABLE $table ADD $col_query;";
 533:             } else {
 534:                 $cmd = array(
 535:                     'mysql|pgsql|mssql|sybase|dblib|odbc|sqlsrv' =>
 536:                         "ALTER TABLE $table ADD $col_query;",
 537:                     'ibm' =>
 538:                         "ALTER TABLE $table ADD COLUMN $col_query;",
 539:                 );
 540:                 $this->queries[] = $this->findQuery($cmd);
 541:             }
 542:         }
 543:         if ($sqlite)
 544:             if ($rebuild || !empty($this->rebuild_cmd)) $this->_sqlite_rebuild($exec);
 545:             else $this->queries += $sqlite_queries;
 546:         $this->queries = array_merge($this->queries,$additional_queries);
 547:         // add new indexes
 548:         foreach ($this->columns as $cname => $column)
 549:             if ($column->index)
 550:                 $this->addIndex($cname, $column->unique);
 551:         if (empty($this->queries))
 552:             return false;
 553:         if (is_array($this->queries) && count($this->queries) == 1)
 554:             $this->queries = $this->queries[0];
 555:         if (!$exec) return $this->queries;
 556:         $result = ($this->suppress)
 557:             ? @$this->db->exec($this->queries) : $this->db->exec($this->queries);
 558:         $this->queries = $this->columns = $this->rebuild_cmd = array();
 559:         return $result;
 560:     }

 562:     /**
 563:      * rebuild a sqlite table with additional schema changes
 564:      */
 565:     protected function _sqlite_rebuild($exec=true)
 566:     {
 567:         $new_columns = $this->columns;
 568:         $existing_columns = $this->getCols(true);
 569:         // find after sorts
 570:         $after = array();
 571:         foreach ($new_columns as $cname => $column)
 572:             if(!empty($column->after))
 573:                 $after[$column->after][] = $cname;
 574:         // find rename commands
 575:         $rename = (!empty($this->rebuild_cmd) && array_key_exists('rename',$this->rebuild_cmd))
 576:                   ? $this->rebuild_cmd['rename'] : array();
 577:         // get primary-key fields
 578:         foreach ($existing_columns as $key => $col)
 579:             if ($col['pkey'])
 580:                 $pkeys[array_key_exists($key,$rename) ? $rename[$key] : $key] = $col;
 581:         foreach ($new_columns as $key => $col)
 582:             if ($col->pkey)
 583:                 $pkeys[$key] = $col;
 584:         // indexes
 585:         $indexes = $this->listIndex();
 586:         // drop fields
 587:         if (!empty($this->rebuild_cmd) && array_key_exists('drop', $this->rebuild_cmd))
 588:             foreach ($this->rebuild_cmd['drop'] as $name)
 589:                 if (array_key_exists($name, $existing_columns)) {
 590:                     if (array_key_exists($name, $pkeys)) {
 591:                         unset($pkeys[$name]);
 592:                         // drop composite key
 593:                         if(count($pkeys) == 1) {
 594:                             $incrementTrigger = $this->db->quotekey($this->name.'_insert');
 595:                             $this->queries[] = 'DROP TRIGGER IF EXISTS '.$incrementTrigger;
 596:                         }
 597:                     }
 598:                     unset($existing_columns[$name]);
 599:                     // drop index
 600:                     foreach (array_keys($indexes) as $col) {
 601:                         // for backward compatibility
 602:                         if ($col == $name)
 603:                             unset($indexes[$name]);
 604:                         // new index names
 605:                         if ($col == $this->name.'___'.$name)
 606:                             unset($indexes[$this->name.'___'.$name]);
 607:                         // check if column is part of an existing combined index
 608:                         if (is_int(strpos($col, '__'))) {
 609:                             if (is_int(strpos($col, '___'))) {
 610:                                 $col = explode('___', $col);
 611:                                 $ci = explode('__', $col[1]);
 612:                                 $col = implode('___',$col);
 613:                             } else // for backward compatibility
 614:                                 $ci = explode('__', $col);
 615:                             // drop combined index
 616:                             if (in_array($name, $ci))
 617:                                 unset($indexes[$col]);
 618:                         }
 619:                     }
 620:                 }       
 621:         // create new table
 622:         $oname = $this->name;
 623:         $this->queries[] = $this->rename($oname.'_temp', false);
 624:         $newTable = $this->schema->createTable($oname);
 625:         // add existing fields
 626:         foreach ($existing_columns as $name => $col) {
 627:             $colName = array_key_exists($name, $rename) ? $rename[$name] : $name;
 628:             // update column datatype
 629:             if (array_key_exists('update',$this->rebuild_cmd)
 630:                 && in_array($name,array_keys($this->rebuild_cmd['update'])))
 631:                 $col['type']=$this->rebuild_cmd['update'][$name];
 632:             $newTable->addColumn($colName, $col)->passThrough();
 633:             // add new fields with after flag
 634:             if (array_key_exists($name,$after))
 635:                 foreach (array_reverse($after[$name]) as $acol) {
 636:                     $newTable->addColumn($new_columns[$acol]);
 637:                     unset($new_columns[$acol]);
 638:                 }
 639:         }
 640:         // add remaining new fields
 641:         foreach ($new_columns as $ncol)
 642:             $newTable->addColumn($ncol);
 643:         $newTable->primary(array_keys($pkeys));
 644:         // add existing indexes
 645:         foreach (array_reverse($indexes) as $name=>$conf) {
 646:             if (is_int(strpos($name, '___')))
 647:                list($tname,$name) = explode('___', $name);
 648:             if (is_int(strpos($name, '__')))
 649:                 $name = explode('__', $name);
 650:             if ($exec) {
 651:                 $t = $this->schema->alterTable($oname);
 652:                 $t->dropIndex($name);
 653:                 $t->build();
 654:             }
 655:             $newTable->addIndex($name,$conf['unique']);
 656:         }
 657:         // build new table
 658:         $newTableQueries = $newTable->build(false);
 659:         $this->queries = array_merge($this->queries,$newTableQueries);
 660:         // copy data
 661:         if (!empty($existing_columns)) {
 662:             foreach (array_keys($existing_columns) as $name) {
 663:                 $fields_from[] = $this->db->quotekey($name);
 664:                 $toName = array_key_exists($name, $rename) ? $rename[$name] : $name;
 665:                 $fields_to[] = $this->db->quotekey($toName);
 666:             }
 667:             $this->queries[] =
 668:                 'INSERT INTO '.$this->db->quotekey($newTable->name).' ('.implode(', ', $fields_to).') '.
 669:                 'SELECT '.implode(', ', $fields_from).' FROM '.$this->db->quotekey($this->name).';';
 670:         }
 671:         $this->queries[] = $this->drop(false);
 672:         $this->name = $oname;
 673:     }

 675:     /**
 676:      * create an insert trigger to work-a-round auto-incrementation in composite primary keys
 677:      * @param $pkey
 678:      * @return array
 679:      */
 680:     public function _sqlite_increment_trigger($pkey) {
 681:         $table = $this->db->quotekey($this->name);
 682:         $pkey = $this->db->quotekey($pkey);
 683:         $triggerName = $this->db->quotekey($this->name.'_insert');
 684:         $queries[] = "DROP TRIGGER IF EXISTS $triggerName;";
 685:         $queries[] = 'CREATE TRIGGER '.$triggerName.' AFTER INSERT ON '.$table.
 686:                      ' WHEN (NEW.'.$pkey.' IS NULL) BEGIN'.
 687:                      ' UPDATE '.$table.' SET '.$pkey.' = ('.
 688:                      ' select coalesce( max( '.$pkey.' ), 0 ) + 1 from '.$table.
 689:                      ') WHERE ROWID = NEW.ROWID;'.
 690:                      ' END;';
 691:         return $queries;
 692:     }

 694:     /**
 695:      * get columns of a table
 696:      * @param bool $types
 697:      * @return array
 698:      */
 699:     public function getCols($types = false)
 700:     {
 701:         $schema = $this->db->schema($this->name, null, 0);
 702:         if (!$types)
 703:             return array_keys($schema);
 704:         else
 705:             foreach ($schema as $name => &$cols) {
 706:                 $default = ($cols['default'] === '') ? null : $cols['default'];
 707:                 if (!is_null($default) && (
 708:                     (is_int(strpos($curdef=$this->findQuery($this->schema->defaultTypes['CUR_STAMP']),
 709:                     $default)) || is_int(strpos($default,$curdef)))
 710:                     || $default == "('now'::text)::timestamp(0) without time zone"))
 711:                 {
 712:                     $default = 'CUR_STAMP';
 713:                 } elseif (!is_null($default)) {
 714:                     // remove single-qoutes
 715:                     if (preg_match('/sqlite2?/', $this->db->driver()))
 716:                         $default=preg_replace('/^\s*([\'"])(.*)\1\s*$/','\2',$default);
 717:                     elseif (preg_match('/mssql|sybase|dblib|odbc|sqlsrv/', $this->db->driver()))
 718:                         $default=preg_replace('/^\s*(\(\')(.*)(\'\))\s*$/','\2',$default);
 719:                     // extract value from character_data in postgre
 720:                     elseif (preg_match('/pgsql/', $this->db->driver()))
 721:                         if (is_int(strpos($default, 'nextval')))
 722:                             $default = null; // drop autoincrement default
 723:                         elseif (preg_match("/^\'*(.*)\'*::(\s*\w)+/", $default, $match))
 724:                             $default = $match[1];
 725:                 } else
 726:                     $default=false;
 727:                 $cols['default'] = $default;
 728:             }
 729:         return $schema;
 730:     }

 736:      */
 737:     public function dropColumn($name)
 738:     {
 739:         $colTypes = $this->getCols(true);
 740:         // check if column exists
 741:         if (!in_array($name, array_keys($colTypes))) return true;

 743:             // SQlite does not support drop column directly
 744:             $this->rebuild_cmd['drop'][] = $name;
 745:         } else {
 746:             $quotedTable = $this->db->quotekey($this->name);
 747:             $quotedColumn = $this->db->quotekey($name);

 749:                 'mysql' =>
 750:                     "ALTER TABLE $quotedTable DROP $quotedColumn;",
 751:                 'pgsql|odbc|ibm|mssql|sybase|dblib|sqlsrv' =>
 752:                     "ALTER TABLE $quotedTable DROP COLUMN $quotedColumn;",
 753:             );
 754:             if (preg_match('/mssql|sybase|dblib|sqlsrv/', $this->db->driver()))
 755:                 $this->suppress=true;
 756:             $this->queries[] = $this->findQuery($cmd);

 760:     /**
 761:      * rename a column
 762:      * @param $name
 763:      * @param $new_name
 764:      * @return void
 765:      */
 766:     public function renameColumn($name, $new_name)
 767:     {
 768:         $existing_columns = $this->getCols(true);
 769:         // check if column is already existing
 770:         if (!in_array($name, array_keys($existing_columns)))
 771:             trigger_error('cannot rename column. it does not exist.');
 772:         if (in_array($new_name, array_keys($existing_columns)))
 773:             trigger_error('cannot rename column. new column already exist.');
 774:         
 775:         if (preg_match('/sqlite2?/', $this->db->driver()))
 776:             // SQlite does not support drop or rename column directly
 777:             $this->rebuild_cmd['rename'][$name] = $new_name;
 778:         elseif (preg_match('/odbc/', $this->db->driver())) {
 779:             // no rename column for odbc, create temp column
 780:             $this->addColumn($new_name, $existing_columns[$name])->passThrough();
 781:             $this->queries[] = "UPDATE $this->name SET $new_name = $name";
 782:             $this->dropColumn($name);
 783:         } else {
 784:             $existing_columns = $this->getCols(true);
 785:             $quotedTable = $this->db->quotekey($this->name);
 786:             $quotedColumn = $this->db->quotekey($name);
 787:             $quotedColumnNew = $this->db->quotekey($new_name);
 788:             $cmd = array(
 789:                 'mysql' =>
 790:                     "ALTER TABLE $quotedTable CHANGE $quotedColumn $quotedColumnNew ".$existing_columns[$name]['type'].";",
 791:                 'pgsql|ibm' =>
 792:                     "ALTER TABLE $quotedTable RENAME COLUMN $quotedColumn TO $quotedColumnNew;",
 793:                 'mssql|sybase|dblib|sqlsrv' =>
 794:                     "sp_rename [$this->name.$name], '$new_name', 'Column'",
 795:             );
 796:             if (preg_match('/mssql|sybase|dblib|sqlsrv/', $this->db->driver()))
 797:                 $this->suppress = true;
 798:             $this->queries[] = $this->findQuery($cmd);
 799:         }
 800:     }

 871:     /**
 872:      * returns table indexes as assoc array
 873:      * @return array
 874:      */
 875:     public function listIndex()
 876:     {
 877:         $table = $this->db->quotekey($this->name);
 878:         $cmd = array(
 879:             'sqlite2?' =>
 880:                 "PRAGMA index_list($table);",
 881:             'mysql' =>
 882:                 "SHOW INDEX FROM $table;",
 883:             'mssql|sybase|dblib|sqlsrv' =>
 884:                 "select * from sys.indexes ".
 885:                 "where object_id = (select object_id from sys.objects where name = '$this->name')",
 886:             'pgsql' =>
 887:                 "select i.relname as name, ix.indisunique as unique ".
 888:                 "from pg_class t, pg_class i, pg_index ix ".
 889:                 "where t.oid = ix.indrelid and i.oid = ix.indexrelid ".
 890:                 "and t.relkind = 'r' and t.relname = '$this->name'".
 891:                 "group by t.relname, i.relname, ix.indisunique;",
 892:         );
 893:         $result = $this->db->exec($this->findQuery($cmd));
 894:         $indexes = array();
 895:         if (preg_match('/pgsql|sqlite2?/', $this->db->driver())) {
 896:             foreach($result as $row)
 897:                 $indexes[$row['name']] = array('unique' => $row['unique']);
 898:         } elseif (preg_match('/mssql|sybase|dblib|sqlsrv/', $this->db->driver())) {
 899:             foreach ($result as $row)
 900:                 $indexes[$row['name']] = array('unique' => $row['is_unique']);
 901:         } elseif (preg_match('/mysql/', $this->db->driver())) {
 902:             foreach($result as $row)
 903:                 $indexes[$row['Key_name']] = array('unique' => !(bool)$row['Non_unique']);
 904:         } else
 905:             trigger_error(sprintf(self::TEXT_ENGINE_NOT_SUPPORTED, $this->db->driver()));
 906:         return $indexes;
 907:     }
